package challenge_8_12;

import java.util.Arrays;
import java.util.HashMap;

public class MiscellaneousStringOperations {

    /**
     * STATIC METHODS
     *
     *  - WordCount
     *  - ArrayToString
     *  - MostFrequent
     *  - ReplaceSubstring
     *
     */

    public static void main(String[] args) {


        String string1 = "the dog  jumped over the fence";
        String string2 = "the";
        String string3 = "that";

        // that dog  jumped over that fence

        System.out.println(getReplaceSubstring(string1, string2, string3));


    }
    static long getWordCount(String words){

        return Arrays.stream(words.split(" ")).count();
    }
    static String getArrayToString(char[] charArray){


        return String.valueOf(charArray);
    }
    static char getMostFrequent(String s) {

        String mostUsedChar = "";

        int count = 0;
        String[] array = s.split("");

        for (int i = 0; i < array.length; i++) {
            int tempCount = 0;

            for (int j = 0; j < array.length; j++)
            {

                if (array[i].equals(array[j])) {
                    tempCount++;
                }
                if (tempCount > count) {
                    count = tempCount;
                    mostUsedChar = array[i];
                }
            }
        }

        return mostUsedChar.charAt(0);

    }
    static String  getReplaceSubstring(String str1, String str2, String str3){

        return str1.replace(str2, str3);

    }



}
